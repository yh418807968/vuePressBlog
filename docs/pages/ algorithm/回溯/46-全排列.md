```js
var permute = function(nums) {
    if(nums.length === 0) return []
    let res = []
    let depth = 0
    let path = []
    let used = []
    dfs(nums, 0, path, used, res)
    return res
};
function dfs(nums, depth, path, used, res) {
   let length = nums.length
   if (length === depth) {
        res.push([...path])
       return
   }
   for (let i = 0; i< length; i++) {
        if (!used[i]) {
            path.push(nums[i])
            used[i] = true
            dfs(nums, depth+1, path, used, res)
            path.pop()
            used[i] = false
        }
   } 
}
```
